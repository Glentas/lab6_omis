DIAGRAM1:
Вот подробное текстовое описание диаграммы состояний интерфейса на основе изображения в формате UML.

```
@startuml
' ======== Диаграмма состояний интерфейса системы ========

' Начальное состояние
[*] --> Login

' Состояние "Login" (Вход в систему)
state Login {
  [*] --> Login
  Login --> Dashboard : Успешная авторизация
  Login --> Login : Неверные учетные данные
}

' Состояние "Dashboard" (Главная панель управления)
state Dashboard {
  [*] --> Dashboard
  Dashboard --> UploadDocument : Загрузить работу
  Dashboard --> ViewHistory : Просмотр истории
  Dashboard --> Settings : Настройки профиля
  Dashboard --> Logout : Выход
  Dashboard --> ViewReport : Просмотр отчета
}

' Состояние "UploadDocument" (Загрузка документа)
state UploadDocument {
  [*] --> UploadDocument
  UploadDocument --> PreprocessingWait : Документ загружен
  UploadDocument --> UploadDocument : Ошибка формата
  UploadDocument --> Dashboard : Отмена
}

' Состояние "PreprocessingWait" (Ожидание предобработки)
state PreprocessingWait {
  [*] --> PreprocessingWait
  PreprocessingWait --> AnalysisWait : Предобработка завершена
  PreprocessingWait --> Dashboard : Переход на главную
}

' Состояние "AnalysisWait" (Ожидание анализа)
state AnalysisWait {
  [*] --> AnalysisWait
  AnalysisWait --> ReportReady : Анализ завершен
  AnalysisWait --> Dashboard : Назад
}

' Состояние "ReportReady" (Отчет готов к просмотру)
state ReportReady {
  [*] --> ReportReady
  ReportReady --> ViewReport : Просмотр результатов
  ReportReady --> UploadDocument : Загрузить новую работу
  ReportReady --> Dashboard : Переход на главную
}

' Состояние "ViewReport" (Просмотр отчета)
state ViewReport {
  [*] --> ViewReport
  ViewReport --> ExportReport : Экспорт в PDF
  ViewReport --> Dashboard : Назад
  ViewReport --> UploadDocument : Проверить новый документ
}

' Состояние "ExportReport" (Экспорт отчета)
state ExportReport {
  [*] --> ExportReport
  ExportReport --> ViewReport : Готово
}

' Состояние "ViewHistory" (Просмотр истории проверок)
state ViewHistory {
  [*] --> ViewHistory
  ViewHistory --> Dashboard : Назад
  ViewHistory --> ViewReport : Выбрать проверку
}

' Состояние "Settings" (Настройки профиля)
state Settings {
  [*] --> Settings
  Settings --> Dashboard : Сохранить
  Settings --> Dashboard : Отмена
}

' Состояние "Logout" (Выход из системы)
state Logout {
  [*] --> Logout
  Logout --> Login
}

@enduml
```


DIAGRAM2:
Вот подробное текстовое описание диаграммы состояний интерфейса на основе изображения в формате UML. Эта диаграмма расширяет предыдущую, добавляя функциональность для роли "Преподаватель/Рецензент".

```
@startuml
' ======== Диаграмма состояний интерфейса системы (Расширенная версия) ========

' Начальное состояние
[*] --> Login

state Login {

  Login --> Login : Неверные учетные данные
}

' Состояние "Dashboard" (Главная панель управления)
state Dashboard {
  [*] --> Dashboard
  ' Общие действия для всех ролей
  Dashboard --> UploadDocument : Загрузить документ
  Dashboard --> ViewReports : Просмотр отчетов
  Dashboard --> Settings : Настройки профиля
  Dashboard --> Logout : Выход

  ' Действия, специфичные для преподавателя
  Dashboard --> ManageStudents : Управление студентами
  Dashboard --> Statistics : Статистика группы
}

' Состояние "UploadDocument" (Загрузка документа)
state UploadDocument {
  [*] --> UploadDocument
  UploadDocument --> DocumentForm : Форма загрузки
  UploadDocument --> Dashboard : Отмена
}

' Состояние "DocumentForm" (Форма загрузки)
state DocumentForm {
  [*] --> DocumentForm
  DocumentForm --> PreprocessingWait : Документ загружен
  DocumentForm --> DocumentForm : Ошибка формата
  DocumentForm --> Dashboard : На главную
}

' Состояние "PreprocessingWait" (Ожидание предобработки)
state PreprocessingWait {
  [*] --> PreprocessingWait
  PreprocessingWait --> AnalysisWait : Предобработка завершена
  PreprocessingWait --> Dashboard : На главную
}

' Состояние "AnalysisWait" (Ожидание анализа)
state AnalysisWait {
  [*] --> AnalysisWait
  AnalysisWait --> ReportReady : Анализ завершен
  AnalysisWait --> Dashboard : Назад
}

' Состояние "ReportReady" (Отчет готов к просмотру)
state ReportReady {
  [*] --> ReportReady
  ReportReady --> ViewReport : Просмотр результатов
  ReportReady --> UploadDocument : Загрузить новую работу
  ReportReady --> Dashboard : На главную
}

' Состояние "ViewReport" (Просмотр отчета)
state ViewReport {
  [*] --> ViewReport
  ViewReport --> ExportReport : Экспорт отчета
  ViewReport --> Dashboard : На главную
  ViewReport --> UploadDocument : Проверить новый документ
}

' Состояние "ExportReport" (Экспорт отчета)
state ExportReport {
  [*] --> ExportReport
  ExportReport --> ViewReport : Готово
}

' Состояние "ViewReports" (Просмотр списка отчетов)
state ViewReports {
  [*] --> ViewReports
  ViewReports --> ReportFilter : Фильтры
  ViewReports --> Dashboard : Назад
  ViewReports --> ViewReport : Выбрать отчет
}

' Состояние "ReportFilter" (Фильтрация отчетов)
state ReportFilter {
  [*] --> ReportFilter
  ReportFilter --> ViewReports : На главную
  ReportFilter --> ViewReport : Выбрать отчет
}

' Состояние "Settings" (Настройки профиля)
state Settings {
  [*] --> Settings
  Settings --> Dashboard : Сохранить
  Settings --> Dashboard : Отмена
}

' Состояние "Logout" (Выход из системы)
state Logout {
  [*] --> Logout
  Logout --> Login
}

' Состояние "ManageStudents" (Управление студентами)
state ManageStudents {
  [*] --> ManageStudents
  ManageStudents --> StudentList : Открыть список
  ManageStudents --> Dashboard : Назад
}

' Состояние "StudentList" (Список студентов)
state StudentList {
  [*] --> StudentList
  StudentList --> StudentDetail : Выбрать студента
  StudentList --> ManageStudents : Назад
}

' Состояние "StudentDetail" (Детали студента)
state StudentDetail {
  [*] --> StudentDetail
  StudentDetail --> ViewStudentReports : Просмотр работ
  StudentDetail --> GradeWork : Оценка работы
  StudentDetail --> SubmitGrade : Выставить оценку
  StudentDetail --> Dashboard : На главную
  StudentDetail --> ManageStudents : Назад
}

' Состояние "ViewStudentReports" (Просмотр работ студента)
state ViewStudentReports {
  [*] --> ViewStudentReports
  ViewStudentReports --> ViewReport : Выбрать отчет
  ViewStudentReports --> StudentDetail : Назад
}

' Состояние "GradeWork" (Оценка работы)
state GradeWork {
  [*] --> GradeWork
  GradeWork --> SubmitGrade : Сохранено
  GradeWork --> StudentDetail : Отмена
}

' Состояние "SubmitGrade" (Выставление оценки)
state SubmitGrade {
  [*] --> SubmitGrade
  SubmitGrade --> StudentDetail : Готово
}

' Состояние "Statistics" (Статистика группы)
state Statistics {
  [*] --> Statistics
  Statistics --> Dashboard : Назад
}

@enduml
```
